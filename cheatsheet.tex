    \section*{Vim Cheat Sheet}

    Keyboard shortcuts for Vim

    \subsection*{General}
    \vspace{-5pt} \titlerule[0.8pt] \vspace{5pt}

    \begin{tabular}{p{0.09\textwidth} p{0.21\textwidth}}
        \textbf{:h[elp] keyword}        & Open documentation for keyword \\
        \textbf{:e[dit] file}           & Open file \\
        \textbf{:w[rite] [file]}        & Save file [as] \\
        \textbf{:q[uit]}                & Close current pane \\
        \textbf{:q[uit]!}               & Close and throw away unsaved changes \\
        \textbf{:wqa}                   & Write and quit all files
    \end{tabular}

    \subsection*{Navigation and Cursor Movement}
    \vspace{-5pt} \titlerule[0.8pt] \vspace{5pt}

    \begin{tabular}{p{0.09\textwidth} p{0.21\textwidth}}
        \textbf{h, j, k,} \Ell{}        & Move cursor left, down, up, or right \\
        \textbf{H, M, L}                & Move to top, middle, bottom of screen \\
        \textbf{w, W}                   & Jump to start of word (cap: including punctuation) \\
        \textbf{e, E}                   & Jump to end of word (cap: including punctuation) \\
        \textbf{b, B}                   & Jump backwards to start of word (cap: including punctuation) \\
        \textbf{0}                      & Jump to start of line \\
        \textbf{\^{}}                   & Jump to first non-black character of line \\
        \textbf{\$}                     & Jump to end of line \\
        \textbf{Ctrl\,+\,f}             & Move down one screen \\
        \textbf{Ctrl\,+\,b}             & Move up one screen \\
        \textbf{Ctrl\,+\,u}             & Move up half a screen \\
        \textbf{Ctrl\,+\,d}             & Move down half a screen \\
    \end{tabular}

    \subsection*{Insert Mode}
    \vspace{-5pt} \titlerule[0.8pt] \vspace{5pt}

    \begin{tabular}{p{0.09\textwidth} p{0.21\textwidth}}
        \textbf{i}                      & Enter insert mode \\
        \textbf{a}                      & Append after the cursor \\
        \textbf{A}                      & Append at end of current line \\
        \textbf{o, O}                   & Append below, above line \\
        \textbf{Esc or Ctrl\,+\,\lbrack}& Exit insert mode \\
        \textbf{Ctrl\,+\,o}             & Exit insert mode for one command \\
        \textbf{Ctrl\,+\,o, hjkl}       & Move left, up, down, right in insert mode \\
        \textbf{Ctrl\,+\,j}             & New line in insert mode \\
        \textbf{Ctrl\,+\,h}             & Backspace in insert mode \\
        \textbf{Ctrl\,+\,w}             & Delete word before cursor in insert mode \\
        \textbf{Ctrl\,+\,n}             & Auto-complete next word in insert mode \\
        \textbf{Ctrl\,+\,p}             & Auto-complete previous word in insert mode \\
        \textbf{Ctrl\,+\,t}             & Indent current line \\
        \textbf{Ctrl\,+\,d}             & De-indent current line \\
    \end{tabular}

    \subsection*{Basic Editing}
    \vspace{-5pt} \titlerule[0.8pt] \vspace{5pt}

    \begin{tabular}{p{0.09\textwidth} p{0.21\textwidth}}

        \textbf{R}                      & Replace multiple characters until exit \\
        \textbf{u}                      & Undo last action \\
        \textbf{U}                      & Restore last undo \\
